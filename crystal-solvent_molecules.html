Диаграммы рассеяния восьми пар разностей дескрипторов кристаллов (горизонтальные оси) и растворителя (вертикальные оси) с наименьшими скорректированными достигаемыми уровнями значимости коэффициента корреляции Спирмена в подвыборке неконформационных полиморфов. Наверху каждой диаграммы рассеяния показаны количество рассматриваемых пар полиморфов (N), скорректированное значение достигаемого уровня значимости (p) коэффициента корреляции Спирмена, а также его величина (ρ)

N = 8
fig, ax = plt.subplots(nrows=2, ncols=4, figsize=(24, 12))

for i, (_, (d1, d2, p_val)) in enumerate(corr_df.sort_values(by=['p_value_corr', 'p_value'], ascending=True).head(N)[['crystal desc', 'solvent desc', 'p_value_corr']].iterrows()):
    axis = ax[i//4, i%4]

    c_df = correlation_df[[d1, d2]].dropna()

    axis.scatter(c_df[d1], c_df[d2], s=500, alpha=.2, color='navy', edgecolor='gold')
    axis.hlines(0, *axis.get_xlim(), colors='k', linestyles='dashed')
    axis.vlines(0, *axis.get_ylim(), colors='k', linestyles='dashed')

    axis.set_title(r'N: {}, p: {:.1g}, $\rho$: {:.3f}'.format(c_df.shape[0], p_val,
                                                              spearmanr(c_df[d1], c_df[d2])[0]),
                   fontdict=dict(fontsize=24))
    axis.set_xlabel(r'$\Delta$' + d1.replace('_px', ''), fontdict=dict(fontsize=24))
    axis.set_ylabel(r'$\Delta$' + d2, fontdict=dict(fontsize=24))
