Диаграммы рассеяния двенадцать пар разностей дескрипторов молекул в кристалле (горизонтальные оси) и растворителя (вертикальные оси) с наименьшими скорректированными достигаемыми уровнями значимости коэффициента корреляции Спирмена в подвыборке неконформационных полиморфов. Наверху каждой диаграммы рассеяния показаны количество рассматриваемых пар полиморфов (N), скорректированное значение достигаемого уровня значимости (p) коэффициента корреляции Спирмена, а также его величина (ρ)

for i, (_, (d1, d2)) in enumerate(corr_df.sort_values(by=['p_value_corr', 'p_value'], ascending=True).head(N)[['crystal molecule desc', 'solvent desc']].iterrows()):
    axis = ax[i//3, i%3]

    c_df = correlation_df[[d1, d2]].dropna()

    axis.scatter(c_df[d1], c_df[d2], s=200, alpha=.45, color='navy', edgecolor='gold')
    axis.hlines(0, *axis.get_xlim(), colors='k', linestyles='dashed')
    axis.vlines(0, *axis.get_ylim(), colors='k', linestyles='dashed')

    axis.set_title(r'$\rho$: {:.3f}, $r$: {:.3f}'.format(spearmanr(c_df[d1], c_df[d2])[0],
                                                         pearsonr(c_df[d1], c_df[d2])[0]))
    axis.set_xlabel(r'$\Delta$' + d1.replace('_mol', ''))
